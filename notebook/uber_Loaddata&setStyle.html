fig = plt.figure(figsize=(20, 14), facecolor='#1a1a2e')
fig.suptitle('🚗 UBER NYC RIDE ANALYSIS DASHBOARD — 2014', 
             fontsize=22, fontweight='bold', color='white', y=0.98)

gs = gridspec.GridSpec(3, 3, figure=fig, hspace=0.45, wspace=0.35)

# ── Chart 1: Trips by Hour ──
ax1 = fig.add_subplot(gs[0, :2])
hourly = df.groupby('hour').size().reset_index(name='trips')
colors_hour = ['#e74c3c' if h in range(17,22) else '#3498db' for h in hourly['hour']]
ax1.bar(hourly['hour'], hourly['trips'], color=colors_hour, edgecolor='none')
ax1.set_title('🕐 Trips by Hour of Day  (Red = Peak Hours)', 
              color='white', fontsize=13, fontweight='bold')
ax1.set_xlabel('Hour', color='white')
ax1.set_ylabel('Trips', color='white')
ax1.set_xticks(range(0,24))

# ── Chart 2: Trips by Month ──
ax2 = fig.add_subplot(gs[0, 2])
month_order = ['April','May','June','July','August','September']
monthly = df.groupby('month_name').size().reset_index(name='trips')
monthly['month_name'] = pd.Categorical(monthly['month_name'], 
                         categories=month_order, ordered=True)
monthly = monthly.sort_values('month_name')
ax2.plot(monthly['month_name'], monthly['trips'], 
         marker='o', color='#2ecc71', linewidth=2.5, markersize=8)
ax2.fill_between(range(len(monthly)), monthly['trips'], alpha=0.3, color='#2ecc71')
ax2.set_title('📈 Monthly Growth', color='white', fontsize=13, fontweight='bold')
ax2.set_xticklabels([m[:3] for m in month_order], rotation=45)
ax2.set_ylabel('Trips', color='white')

# ── Chart 3: Trips by Day of Week ──
ax3 = fig.add_subplot(gs[1, :2])
day_order = ['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday']
daily = df.groupby('weekday').size().reset_index(name='trips')
daily['weekday'] = pd.Categorical(daily['weekday'], categories=day_order, ordered=True)
daily = daily.sort_values('weekday')
colors_day = ['#e74c3c' if d in ['Friday','Saturday'] else '#9b59b6' for d in daily['weekday']]
ax3.bar(daily['weekday'], daily['trips'], color=colors_day, edgecolor='none')
ax3.set_title('📅 Trips by Day of Week  (Red = Busiest Days)', 
              color='white', fontsize=13, fontweight='bold')
ax3.set_xlabel('Day', color='white')
ax3.set_ylabel('Trips', color='white')
ax3.set_xticklabels(day_order, rotation=30)

# ── Chart 4: Trips by Base ──
ax4 = fig.add_subplot(gs[1, 2])
base = df.groupby('base').size().reset_index(name='trips').sort_values('trips')
ax4.barh(base['base'], base['trips'], color='#e67e22', edgecolor='none')
ax4.set_title('🚗 Trips by Base', color='white', fontsize=13, fontweight='bold')
ax4.set_xlabel('Trips', color='white')

# ── Chart 5: Heatmap Hour vs Day ──
ax5 = fig.add_subplot(gs[2, :])
pivot = df.groupby(['weekday','hour']).size().unstack()
pivot = pivot.reindex(day_order)
sns.heatmap(pivot, ax=ax5, cmap='YlOrRd', linewidths=0.3,
            cbar_kws={'label':'Trips', 'shrink':0.8})
ax5.set_title('🔥 Heatmap — Trips by Hour & Day of Week', 
              color='white', fontsize=13, fontweight='bold')
ax5.set_xlabel('Hour of Day', color='white')
ax5.set_ylabel('Day of Week', color='white')

plt.savefig(r'G:\Project\uber-ride-analysis\visuals\MASTER_DASHBOARD.png', 
            dpi=200, bbox_inches='tight', facecolor='#1a1a2e')
plt.show()
print("✅ Master Dashboard saved!")

fig, ax = plt.subplots(figsize=(12,7), facecolor='#1a1a2e')
ax.set_facecolor('#1a1a2e')
ax.axis('off')

# Calculate stats
total_trips     = df.shape[0]
peak_hour       = df.groupby('hour').size().idxmax()
peak_day        = df.groupby('weekday').size().idxmax()
peak_month      = df.groupby('month_name').size().idxmax()
busiest_base    = df.groupby('base').size().idxmax()
avg_daily       = int(df.groupby(df['date_time'].dt.date).size().mean())
growth          = round(((df[df['month_name']=='September'].shape[0] - 
                          df[df['month_name']=='April'].shape[0]) / 
                          df[df['month_name']=='April'].shape[0]) * 100, 1)

insights = [
    ("🚗  Total Trips Analyzed",   f"{total_trips:,}"),
    ("🕐  Peak Booking Hour",       f"{peak_hour}:00 — {'PM' if peak_hour >= 12 else 'AM'} Rush"),
    ("📅  Busiest Day of Week",     f"{peak_day}"),
    ("📆  Busiest Month",           f"{peak_month}"),
    ("📍  Most Active Base",        f"{busiest_base}"),
    ("📊  Avg Daily Trips",         f"{avg_daily:,}"),
    ("📈  Demand Growth (Apr-Sep)", f"+{growth}%"),
]

ax.text(0.5, 0.97, '📊 UBER NYC — KEY INSIGHTS',
        transform=ax.transAxes, fontsize=20, fontweight='bold',
        color='white', ha='center', va='top')

# ✅ Fixed: use ax.plot instead of axhline for transform support
ax.plot([0.05, 0.95], [0.88, 0.88], color='#3498db', 
        linewidth=2, transform=ax.transAxes)

for i, (label, value) in enumerate(insights):
    y_pos = 0.80 - i * 0.10
    ax.text(0.08, y_pos, label, transform=ax.transAxes,
            fontsize=13, color='#bdc3c7', va='center')
    ax.text(0.92, y_pos, value, transform=ax.transAxes,
            fontsize=13, color='#2ecc71', fontweight='bold',
            va='center', ha='right')
    # ✅ Fixed: use ax.plot instead of axhline
    ax.plot([0.05, 0.95], [y_pos - 0.04, y_pos - 0.04], 
            color='#2c3e50', linewidth=0.8, transform=ax.transAxes)

plt.tight_layout()
plt.savefig(r'G:\Project\uber-ride-analysis\visuals\key_insights_card.png',
            dpi=150, bbox_inches='tight', facecolor='#1a1a2e')
plt.show()
print("✅ Insights card saved!")
